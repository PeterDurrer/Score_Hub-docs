Mehrsprachigkeit (i18n) ​

ScoreHub unterstützt drei Sprachen: Deutsch (de), Französisch (fr) und Italienisch (it).

Architektur ​

frontend/src/i18n/
├── config.ts              # i18n Konfiguration
└── locales/
    ├── de.json            # Deutsche Übersetzungen
    ├── fr.json            # Französische Übersetzungen
    └── it.json            # Italienische Übersetzungen

Neue Übersetzung hinzufügen ​

1. Key in allen Sprachdateien hinzufügen ​

json
// de.json
{
  "competitions": {
    "newButton": "Neuer Wettkampf"
  }
}

// fr.json
{
  "competitions": {
    "newButton": "Nouvelle compétition"
  }
}

// it.json
{

2. Im Code verwenden ​

tsx
import { useTranslation } from 'react-i18next';

export function CompetitionsPage() {
  const { t } = useTranslation();
  
  return (
    <button>
      {t('competitions.newButton')}
    </button>
  );
}

Sprachumschaltung ​

Der LanguageSwitcher Component ermöglicht die Sprachwahl:

tsx
import { useTranslation } from 'react-i18next';

export function LanguageSwitcher() {
  const { i18n } = useTranslation();
  
  const languages = [
    { code: 'de', label: 'DE' },
    { code: 'fr', label: 'FR' },
    { code: 'it', label: 'IT' },
  ];
  
  return (
    <div className="flex gap-2">
      {languages.map(({ code, label }) => (
        <button
          key={code}
          onClick={() => i18n.changeLanguage(code)}
          className={i18n.language === code ? 'font-bold' : ''}
        >
          {label}
        </button>
      ))}
    </div>
  );
}

XLF Export/Import ​

Für professionelle Übersetzungsworkflows unterstützen wir XLF (XLIFF) Format.

Export zu XLF ​

powershell
# Script: scripts/export-xlf.js
node scripts/export-xlf.js de fr  # Exportiert de als Quelle, fr als Ziel
javascript
// scripts/export-xlf.js
const fs = require('fs');
const path = require('path');

function jsonToXlf(source, target, sourceLang, targetLang) {
  let units = '';
  
  function processObject(obj, targetObj, prefix = '') {
    for (const key in obj) {
      const fullKey = prefix ? `${prefix}.${key}` : key;
      if (typeof obj[key] === 'object') {
        processObject(obj[key], targetObj?.[key] || {}, fullKey);
      } else {
        units += `
      <trans-unit id="${fullKey}">
        <source>${escapeXml(obj[key])}</source>
        <target>${escapeXml(targetObj?.[key] || obj[key])}</target>
      </trans-unit>`;
      }
    }
  }
  
  processObject(source, target);
  
  return `<?xml version="1.0" encoding="UTF-8"?>
<xliff version="1.2" xmlns="urn:oasis:names:tc:xliff:document:1.2">
  <file source-language="${sourceLang}" target-language="${targetLang}" datatype="plaintext" original="messages">
    <body>${units}
    </body>
  </file>
</xliff>`;
}

function escapeXml(str) {
  return str.replace(/[<>&'"]/g, c => ({
    '<': '&lt;', '>': '&gt;', '&': '&amp;', "'": '&apos;', '"': '&quot;'
  }[c]));
}

// Ausführung
const [,, sourceLang, targetLang] = process.argv;
const sourceFile = require(`../frontend/src/i18n/locales/${sourceLang}.json`);
const targetFile = require(`../frontend/src/i18n/locales/${targetLang}.json`);

const xlf = jsonToXlf(sourceFile, targetFile, sourceLang, targetLang);
fs.writeFileSync(`translations/${sourceLang}-${targetLang}.xlf`, xlf);
console.log(`Exported: translations/${sourceLang}-${targetLang}.xlf`);

Import von XLF ​

powershell
# Script: scripts/import-xlf.js
node scripts/import-xlf.js translations/de-fr.xlf fr
javascript
// scripts/import-xlf.js
const fs = require('fs');
const { DOMParser } = require('xmldom');

function xlfToJson(xlfContent) {
  const doc = new DOMParser().parseFromString(xlfContent, 'text/xml');
  const units = doc.getElementsByTagName('trans-unit');
  const result = {};
  
  for (let i = 0; i < units.length; i++) {
    const id = units[i].getAttribute('id');
    const target = units[i].getElementsByTagName('target')[0];
    const value = target ? target.textContent : '';
    
    // Nested key support: "a.b.c" -> { a: { b: { c: value } } }
    const keys = id.split('.');
    let current = result;
    for (let j = 0; j < keys.length - 1; j++) {
      current[keys[j]] = current[keys[j]] || {};
      current = current[keys[j]];
    }
    current[keys[keys.length - 1]] = value;
  }
  
  return result;
}

// Ausführung
const [,, xlfPath, targetLang] = process.argv;
const xlfContent = fs.readFileSync(xlfPath, 'utf-8');
const json = xlfToJson(xlfContent);

fs.writeFileSync(
  `frontend/src/i18n/locales/${targetLang}.json`,
  JSON.stringify(json, null, 2)
);
console.log(`Imported to: frontend/src/i18n/locales/${targetLang}.json`);

Best Practices ​

1. Konsistente Key-Struktur ​

json
{
  "common": {
    "save": "Speichern",
    "cancel": "Abbrechen",
    "delete": "Löschen",
    "loading": "Wird geladen..."
  },
  "pages": {
    "home": { ... },
    "competitions": { ... }
  },
  "forms": {
    "validation": {
      "required": "Pflichtfeld",
      "email": "Ungültige E-Mail"
    }
  }
}

2. Keine hartcodierten Strings ​

tsx
// ❌ Schlecht
<h1>Wettkämpfe</h1>

// ✅ Gut
<h1>{t('pages.competitions.title')}</h1>

3. Kontext für Übersetzer ​

json
{
  "save": "Speichern",
  "_save_context": "Button zum Speichern eines Formulars"
}

4. Formatierung ​

Verwenden Sie react-i18next Formatierung für Daten und Zahlen:

tsx
import { Trans } from 'react-i18next';

// Datum
<Trans i18nKey="lastUpdated" values={{ date: new Date() }}>
  Zuletzt aktualisiert: {{ date, formatParams: { date: { dateStyle: 'long' } } }}
</Trans>

// Zahl
t('points', { score: 1234.5, formatParams: { score: { style: 'decimal' } } })

Fehlende Übersetzungen ​

Development Warning ​

In der Entwicklung werden fehlende Keys als Warning angezeigt:

typescript
// i18n/config.ts
i18n.init({
  debug: import.meta.env.DEV,
  saveMissing: import.meta.env.DEV,
  missingKeyHandler: (lng, ns, key) => {
    console.warn(`Missing translation: ${key} [${lng}]`);
  }
});

Fallback Kette ​

typescript
i18n.init({
  fallbackLng: 'de',
  interpolation: {
    escapeValue: false
  }
});

CI/CD Integration ​

Validierung in CI ​

yaml
# .github/workflows/ci.yml
- name: Validate translations
  run: node scripts/validate-translations.js
javascript
// scripts/validate-translations.js
const de = require('../frontend/src/i18n/locales/de.json');
const fr = require('../frontend/src/i18n/locales/fr.json');
const it = require('../frontend/src/i18n/locales/it.json');

function getKeys(obj, prefix = '') {
  let keys = [];
  for (const key in obj) {
    const fullKey = prefix ? `${prefix}.${key}` : key;
    if (typeof obj[key] === 'object') {
      keys = [...keys, ...getKeys(obj[key], fullKey)];
    } else {
      keys.push(fullKey);
    }
  }
  return keys;
}

const deKeys = new Set(getKeys(de));
const frKeys = new Set(getKeys(fr));
const itKeys = new Set(getKeys(it));

let hasErrors = false;

// Check French
for (const key of deKeys) {
  if (!frKeys.has(key)) {
    console.error(`Missing FR: ${key}`);
    hasErrors = true;
  }
}

// Check Italian
for (const key of deKeys) {
  if (!itKeys.has(key)) {
    console.error(`Missing IT: ${key}`);
    hasErrors = true;
  }
}

if (hasErrors) {
  process.exit(1);
}

console.log('All translations complete!');
